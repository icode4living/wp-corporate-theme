    function custom_get_excerpt($post_id){
        $temp = $post;
        $post = get_post( $post_id);
        setup_postdata( $post );
        $excerpt = esc_attr( strip_tags(get_the_excerpt()));
        wp_reset_postdata();
        $post = $temp;
            
        return $excerpt;
    }
